Quick Example

# Train a CNN on MNIST
NN-CLI --config mnist_cnn_config.json \
       --mode train \
       --idx-data train-images.idx3 \
       --idx-labels train-labels.idx1

# Predict with the trained model
NN-CLI --config trained_model.json \
       --mode predict \
       --input digit.json
